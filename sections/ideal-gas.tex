\begin{cheatformula}[חוק הגז האידאלי]
\begin{center}
\begin{tabular}{lr}
    $\displaystyle P\,V = N\,k_{B}\,T$ 
    & 
    \scalebox{0.6}{$
    \begin{aligned}
        [P] &= 1\,\mathrm{atm} = 1.013 \times 10^{5}\,\mathrm{Pa} \\

        [T] &= \mathrm{K}
    \end{aligned}$}
\end{tabular}
\end{center}
\end{cheatformula}

\begin{cheatformula}[אנרגיה קינטית ממוצעת של חלקיק גז אידאלי]

\end{cheatformula}

    \begin{align*}

    \end{align*}
    

\end{cheatformula}

\end{align*}


\end{cheatformula}
