\section{תהליכים הפיכים}
כלל אצבע - אם תראה וידאו של תהליך כלשהו ואז מישהו יגיד לך שהוידאו היה בrevese ולא חשדת אז התהליך הפיך
\begin{cheatformula}[תהליך הפיך]
    $$dS = \frac{\dbar Q}{T}$$
\end{cheatformula}

\begin{cheatformula}[תהליך בלתי-הפיך]
    $$dS > \frac{\dbar Q}{T}$$
\end{cheatformula}

\begin{cheatformula}[מכונת חום]
    חום נכנס - $Q_H$ \\
    חום יוצא - $Q_C$
    $$Q_H = \abs{Q_C} + \abs{W} $$
    נצילות של מכונת חום:
    $$\eta_{eng} = \frac{\abs{W}}{Q_H} = 1 - \frac{\abs{Q_C}}{Q_H}$$
\end{cheatformula}
